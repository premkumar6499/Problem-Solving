C program for converting Roman Letter to Number.....

#include <stdio.h>
#include <string.h>

int romanValue(char c) {
    switch (c) {
        case 'I': return 1;
        case 'V': return 5;
        case 'X': return 10;
        case 'L': return 50;
        case 'C': return 100;
        case 'D': return 500;
        case 'M': return 1000;
        default: return 0;
    }
}

int romanToInt(char* s) {
    int total = 0;
    int length = strlen(s);
    
    for (int i = 0; i < length; i++) {
        int current = romanValue(s[i]);
        
        // Peek at the next character
        if (i + 1 < length) {
            int next = romanValue(s[i + 1]);
            
            // Subtraction rule: if current < next, subtract current
            if (current < next) {
                total -= current;
            } else {
                total += current;
            }
        } else {
            // Last character is always added
            total += current;
        }
    }
    
    return total;
}

int main() {
    char* example = "MCMXCIV";
    printf("Input: %s\nOutput: %d\n", example, romanToInt(example));
    return 0;
}
